6) Скрипт №3 — заполнить справочники и связать ID

-- Справочники из данных
INSERT INTO roles(name)
SELECT DISTINCT TRIM(role_name) FROM users WHERE TRIM(role_name) <> '';

INSERT INTO order_statuses(name)
SELECT DISTINCT TRIM(status_name) FROM orders WHERE TRIM(status_name) <> '';

INSERT INTO suppliers(name)
SELECT DISTINCT TRIM(supplier_name) FROM products WHERE TRIM(supplier_name) <> '';

INSERT INTO manufacturers(name)
SELECT DISTINCT TRIM(manufacturer_name) FROM products WHERE TRIM(manufacturer_name) <> '';

INSERT INTO categories(name)
SELECT DISTINCT TRIM(category_name) FROM products WHERE TRIM(category_name) <> '';

-- Проставляем ID
UPDATE users u
SET role_id = r.id
FROM roles r
WHERE TRIM(u.role_name) = TRIM(r.name);

UPDATE products p
SET supplier_id = s.id
FROM suppliers s
WHERE TRIM(p.supplier_name) = TRIM(s.name);

UPDATE products p
SET manufacturer_id = m.id
FROM manufacturers m
WHERE TRIM(p.manufacturer_name) = TRIM(m.name);

UPDATE products p
SET category_id = c.id
FROM categories c
WHERE TRIM(p.category_name) = TRIM(c.name);

UPDATE orders o
SET client_id = u.user_id
FROM users u
WHERE TRIM(o.client_full_name) = TRIM(u.full_name);

UPDATE orders o
SET status_id = s.id
FROM order_statuses s
WHERE TRIM(o.status_name) = TRIM(s.name);

-- Разобрать адреса пунктов выдачи
UPDATE pickup_points
SET
    post_index = TRIM(split_part(full_address, ',', 1)),
    city       = TRIM(replace(split_part(full_address, ',', 2), 'г.', '')),
    street     = TRIM(replace(split_part(full_address, ',', 3), 'ул.', '')),
    house      = TRIM(split_part(full_address, ',', 4));


7) Скрипт №4 — состав заказа (чтобы не строкой, а нормально)

CREATE TABLE order_items (
    id SERIAL PRIMARY KEY,
    order_id INT NOT NULL,
    product_article TEXT NOT NULL,
    qty INT NOT NULL CHECK (qty > 0),
    UNIQUE(order_id, product_article)
);

INSERT INTO order_items(order_id, product_article, qty)
SELECT
    o.order_id,
    TRIM(arr[i]) AS product_article,
    TRIM(arr[i+1])::INT AS qty
FROM (
    SELECT order_id, regexp_split_to_array(order_articles, '\s*,\s*') AS arr
    FROM orders
) o
CROSS JOIN LATERAL generate_series(1, array_length(arr,1), 2) AS i;


8) Скрипт №5 — включить ссылочную целостность (FK)
ALTER TABLE users
    ADD CONSTRAINT fk_users_role
    FOREIGN KEY (role_id) REFERENCES roles(id);

ALTER TABLE products
    ADD CONSTRAINT fk_products_supplier
    FOREIGN KEY (supplier_id) REFERENCES suppliers(id),
    ADD CONSTRAINT fk_products_manufacturer
    FOREIGN KEY (manufacturer_id) REFERENCES manufacturers(id),
    ADD CONSTRAINT fk_products_category
    FOREIGN KEY (category_id) REFERENCES categories(id);

ALTER TABLE orders
    ADD CONSTRAINT fk_orders_client
    FOREIGN KEY (client_id) REFERENCES users(user_id),
    ADD CONSTRAINT fk_orders_status
    FOREIGN KEY (status_id) REFERENCES order_statuses(id),
    ADD CONSTRAINT fk_orders_pickup
    FOREIGN KEY (pickup_point_ref) REFERENCES pickup_points(pickup_point_id);

ALTER TABLE order_items
    ADD CONSTRAINT fk_order_items_order
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    ADD CONSTRAINT fk_order_items_product
    FOREIGN KEY (product_article) REFERENCES products(article);
